news_event = {	#Francis Fukuyama - History never Ends 
	id = newsrw.1
	title = newsrw.1.t
	desc = newsrw.1.d
	picture = fukuyama
	
	is_triggered_only = yes
	major = yes
	
	trigger = {
	    date > 2010.02.30
		date < 2010.03.30
	}
	
	option = {
		name = newsrw.1.a
		trigger = { has_government = communist has_government = revolutionary_socialist }
	}

	option = {
		name = newsrw.1.b
		trigger = { NOT = { has_government = communist has_government = revolutionary_socialist } }
	}
}
