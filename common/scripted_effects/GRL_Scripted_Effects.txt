
#Important to put HIGHEST first
GRL_stability_level_up = {
	custom_effect_tooltip = GRL_stability_up_tt
	if = {
		limit = { has_idea = GRL_Stability_Okay }
		swap_ideas = {
			remove_idea = GRL_Stability_Okay 
			add_idea = GRL_Stability_Grand 
		}
	}
	else_if = {
		limit = { has_idea = GRL_Stability_Questionable }
		swap_ideas = {
			remove_idea = GRL_Stability_Questionable 
			add_idea = GRL_Stability_Okay 
		}
	}
	else_if = {
		limit = { has_idea = GRL_Stability_Threatened }
		swap_ideas = {
			remove_idea = GRL_Stability_Threatened 
			add_idea = GRL_Stability_Questionable 
		}
	}
}

#Important to put LOWEST first
GRL_stability_level_down = {
custom_effect_tooltip = GRL_stability_down_tt
	if = {
		limit = { has_idea = GRL_Stability_Questionable }
		swap_ideas = {
			remove_idea = GRL_Stability_Questionable 
			add_idea = GRL_Stability_Threatened 
		}
	}
	if = {
		limit = { has_idea = GRL_Stability_Okay }
		swap_ideas = {
			remove_idea = GRL_Stability_Okay 
			add_idea = GRL_Stability_Questionable 
		}
	}
	if = {
		limit = { has_idea = GRL_Stability_Grand }
		swap_ideas = {
			remove_idea = GRL_Stability_Grand 
			add_idea = GRL_Stability_Okay 
		}
	}
}
